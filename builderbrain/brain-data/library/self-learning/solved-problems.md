```python
import json

def get_top_3_keys(file_path):
    """
    Reads a JSON file and returns the top 3 keys by value.

    Args:
    file_path (str): Path to the JSON file.

    Returns:
    list: A list of tuples containing the top 3 keys and their values.
    """

    # Load the JSON file
    with open(file_path, 'r') as f:
        data = json.load(f)

    # Sort the keys by value in descending order and get the top 3
    top_3 = sorted(data.items(), key=lambda x: x[1], reverse=True)[:3]

    return top_3

# Example usage:
file_path = 'example.json'
top_3_keys = get_top_3_keys(file_path)
print(top_3_keys)
```
